 var dukeImages = new Array("T2.gif","T4.gif","T6.gif","T8.gif","T10.gif")
 var thisDuke=0

 function previousDuke() {

   if (document.images && thisDuke > 0) {

       thisDuke--
       document.seqDuke.src = dukeImages[thisDuke]

  }

 }

 function nextDuke() {

   if (document.images && thisDuke < 5) {

       thisDuke++
       document.seqDuke.src = dukeImages[thisDuke]

  }

 }
